class CQueue {
    stack<int> s1;
    stack<int> s2;
public:
    CQueue() {}
    
    void appendTail(int value) {
        s1.push(value); //push()
    }
    
    int deleteHead() {
      	//empty()
        if(s2.empty()){ 
            if(s1.empty())return -1;
            else{
                while(!s1.empty()){
                  	//需要先读取top()，再pop()，pop()返回值是void，不是弹出的值
                    s2.push(s1.top());   
                    s1.pop();                 
                }
            }
        }
        int top_s2 = s2.top();
        s2.pop();
        return top_s2;
    }    
};
